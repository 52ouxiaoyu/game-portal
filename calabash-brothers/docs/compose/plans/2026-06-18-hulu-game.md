```javascript
// 物理常量
const GRAVITY = 0.6;
const FRICTION = 0.8;
const JUMP_FORCE = -12;

// 实体类
class Entity {
    constructor(x, y, width, height, color) {
        this.x = x;
        this.y = y;
        this.width = width;
        this.height = height;
        this.color = color;
        this.velocityX = 0;
        this.velocityY = 0;
        this.onGround = false;
    }

    applyGravity() {
        this.velocityY += GRAVITY;
    }

    update() {
        this.x += this.velocityX;
        this.y += this.velocityY;
        this.velocityX *= FRICTION;
    }

    draw(ctx) {
        ctx.fillStyle = this.color;
        ctx.fillRect(this.x, this.y, this.width, this.height);
    }
}
```

- [ ] **Step 2: 添加碰撞检测函数**

```javascript
// 碰撞检测
function checkCollision(entity1, entity2) {
    return entity1.x < entity2.x + entity2.width &&
           entity1.x + entity1.width > entity2.x &&
           entity1.y < entity2.y + entity2.height &&
           entity1.y + entity1.height > entity2.y;
}

function resolveCollision(entity, platform) {
    // 检查是否从上方碰撞
    if (entity.velocityY > 0 && 
        entity.y + entity.height - entity.velocityY <= platform.y) {
        entity.y = platform.y - entity.height;
        entity.velocityY = 0;
        entity.onGround = true;
    }
}
```

```javascript
const HULU_BROTHERS = [
    { name: '大娃', color: '#e74c3c', ability: '力大无穷' },
    { name: '二娃', color: '#f39c12', ability: '千里眼' },
    { name: '三娃', color: '#f1c40f', ability: '铜头铁臂' },
    { name: '四娃', color: '#2ecc71', ability: '喷火' },
    { name: '五娃', color: '#1abc9c', ability: '吸水' },
    { name: '六娃', color: '#3498db', ability: '隐身' },
    { name: '七娃', color: '#9b59b6', ability: '宝葫芦' }
];

// 玩家类
class Player extends Entity {
    constructor(x, y, brotherIndex) {
        super(x, y, 24, 32, HULU_BROTHERS[brotherIndex].color);
        this.brotherIndex = brotherIndex;
        this.name = HULU_BROTHERS[brotherIndex].name;
        this.health = 5;
        this.maxHealth = 5;
        this.invincible = false;
        this.invincibleTimer = 0;
    }

    handleInput() {
        // 左右移动
        if (keys.a) {
            this.velocityX = -5;
        } else if (keys.d) {
            this.velocityX = 5;
        }

        // 跳跃
        if (keys.g && this.onGround) {
            this.velocityY = JUMP_FORCE;
            this.onGround = false;
        }
    }

    takeDamage() {
        if (!this.invincible) {
            this.health--;
            this.invincible = true;
            this.invincibleTimer = 60; // 1秒无敌时间
            
            // 更新血条
            const healthPercent = (this.health / this.maxHealth) * 100;
            document.getElementById('healthFill').style.width = healthPercent + '%';
            
            if (this.health <= 0) {
                game.state = 'gameover';
            }
        }
    }

    update() {
        this.handleInput();
        super.update();
        
        // 无敌时间处理
        if (this.invincible) {
            this.invincibleTimer--;
            if (this.invincibleTimer <= 0) {
                this.invincible = false;
            }
        }
        
        // 边界检测
        this.x = Math.max(0, Math.min(CANVAS_SIZE - this.width, this.x));
        this.y = Math.max(0, Math.min(CANVAS_SIZE - this.height, this.y));
    }

    draw(ctx) {
        if (this.invincible && Math.floor(this.invincibleTimer / 4) % 2 === 0) {
            return; // 无敌时闪烁
        }
        
        // 绘制葫芦娃身体
        ctx.fillStyle = this.color;
        ctx.fillRect(this.x + 4, this.y + 8, 16, 20); // 身体
        
        // 绘制葫芦娃头部
        ctx.fillStyle = '#fdbcb4';
        ctx.fillRect(this.x + 6, this.y, 12, 10); // 头部
        
        // 绘制葫芦娃眼睛
        ctx.fillStyle = '#000';
        ctx.fillRect(this.x + 8, this.y + 3, 2, 2); // 左眼
        ctx.fillRect(this.x + 14, this.y + 3, 2, 2); // 右眼
        
        // 绘制葫芦娃嘴巴
        ctx.fillRect(this.x + 10, this.y + 7, 4, 1);
    }
}
```

```javascript
class LevelManager {
    constructor() {
        this.currentLevel = 0;
        this.platforms = [];
        this.enemies = [];
        this.collectibles = [];
        this.exit = null;
        this.boss = null;
    }

    loadLevel(levelIndex) {
        const level = LEVELS[levelIndex];
        this.currentLevel = levelIndex;
        this.platforms = level.platforms.map(p => new Entity(p.x, p.y, p.width, p.height, '#8B4513'));
        this.enemies = level.enemies.map(e => this.createEnemy(e));
        this.collectibles = level.collectibles.map(c => this.createCollectible(c));
        this.exit = level.exit;
        this.boss = level.boss ? this.createBoss(level.boss) : null;
    }

    createEnemy(enemyData) {
        // 敌人创建逻辑
        return new Entity(enemyData.x, enemyData.y, 24, 24, '#ff0000');
    }

    createCollectible(collectibleData) {
        // 收集品创建逻辑
        return new Entity(collectibleData.x, collectibleData.y, 16, 16, '#ffcc00');
    }

    createBoss(bossData) {
        // Boss创建逻辑
        return new Entity(bossData.x, bossData.y, 48, 48, '#800080');
    }

    draw(ctx) {
        // 绘制平台
        this.platforms.forEach(p => p.draw(ctx));
        
        // 绘制敌人
        this.enemies.forEach(e => e.draw(ctx));
        
        // 绘制收集品
        this.collectibles.forEach(c => c.draw(ctx));
        
        // 绘制出口
        if (this.exit) {
            ctx.fillStyle = '#00ff00';
            ctx.fillRect(this.exit.x, this.exit.y, 32, 32);
        }
        
        // 绘制Boss
        if (this.boss) {
            this.boss.draw(ctx);
        }
    }
}
```

```javascript
let player = null;
let levelManager = new LevelManager();

function startGame(brotherIndex) {
    player = new Player(100, CANVAS_SIZE - 100, brotherIndex);
    levelManager.loadLevel(0);
    game.state = 'playing';
    game.score = 0;
    document.getElementById('score').textContent = '分数: 0';

// 游戏循环
function gameLoop() {
    update();
    render();
    requestAnimationFrame(gameLoop);
}

function update() {
    switch (game.state) {
        case 'menu':
            handleMenuInput();
            break;
        case 'playing':
            updatePlaying();
            break;
        case 'gameover':
            // 处理游戏结束
            break;
        case 'victory':
            // 处理胜利
            break;
    }
}

function updatePlaying() {
    // 更新玩家
    player.update();
    
    // 检查平台碰撞
    levelManager.platforms.forEach(platform => {
        if (checkCollision(player, platform)) {
            resolveCollision(player, platform);
        }
    });
    
    // 检查敌人碰撞
    levelManager.enemies.forEach((enemy, index) => {
        if (checkCollision(player, enemy)) {
            // 检查是否踩到敌人
            if (player.velocityY > 0 && 
                player.y + player.height - player.velocityY <= enemy.y) {
                // 消灭敌人
                levelManager.enemies.splice(index, 1);
                game.score += 100;
                document.getElementById('score').textContent = '分数: ' + game.score;
            } else {
                // 受到伤害
                player.takeDamage();
            }
        }
    });
    
    // 检查收集品碰撞
    levelManager.collectibles.forEach((collectible, index) => {
        if (checkCollision(player, collectible)) {
            // 收集道具
            levelManager.collectibles.splice(index, 1);
            game.score += 50;
            document.getElementById('score').textContent = '分数: ' + game.score;
            
            // 如果是葫芦瓶，增加生命
            if (collectible.color === '#ffcc00') {
                player.health = Math.min(player.maxHealth, player.health + 1);
                const healthPercent = (player.health / player.maxHealth) * 100;
                document.getElementById('healthFill').style.width = healthPercent + '%';
            }
        }
    });
    
    // 检查出口
    if (levelManager.exit && checkCollision(player, levelManager.exit)) {
        // 进入下一关
        if (levelManager.currentLevel < LEVELS.length - 1) {
            levelManager.loadLevel(levelManager.currentLevel + 1);
            player.x = 100;
            player.y = CANVAS_SIZE - 100;
        } else {
            game.state = 'victory';
        }
    }
}

function render() {
    switch (game.state) {
        case 'menu':
            renderMenu();
            break;
        case 'playing':
            renderPlaying();
            break;
        case 'gameover':
            renderGameOver();
            break;
        case 'victory':
            renderVictory();
            break;
    }
}
```

```javascript
        if (!this.audioContext) this.init();
        
        const oscillator = this.audioContext.createOscillator();
        const gainNode = this.audioContext.createGain();
        
        oscillator.connect(gainNode);
        gainNode.connect(this.audioContext.destination);
        
        switch (type) {
            case 'jump':
                oscillator.frequency.setValueAtTime(300, this.audioContext.currentTime);
                oscillator.frequency.exponentialRampToValueAtTime(600, this.audioContext.currentTime + 0.1);
                gainNode.gain.setValueAtTime(0.3, this.audioContext.currentTime);
                gainNode.gain.exponentialRampToValueAtTime(0.01, this.audioContext.currentTime + 0.1);
                oscillator.start(this.audioContext.currentTime);
                oscillator.stop(this.audioContext.currentTime + 0.1);
                break;
            case 'collect':
                oscillator.frequency.setValueAtTime(500, this.audioContext.currentTime);
```

```javascript
// 粒子系统
class Particle {
    constructor(x, y, color) {
        this.x = x;
        this.y = y;
        this.color = color;
        this.size = Math.random() * 3 + 1;
        this.speedX = Math.random() * 6 - 3;
        this.speedY = Math.random() * 6 - 3;
        this.life = 1;
        this.decay = Math.random() * 0.02 + 0.01;
    }

    update() {
        this.x += this.speedX;
        this.y += this.speedY;
        this.life -= this.decay;
    }

    draw(ctx) {
        ctx.globalAlpha = this.life;
        ctx.fillStyle = this.color;
        ctx.fillRect(this.x, this.y, this.size, this.size);
        ctx.globalAlpha = 1;
    }
}

class ParticleSystem {
    constructor() {
        this.particles = [];
    }

    emit(x, y, color, count = 10) {
        for (let i = 0; i < count; i++) {
            this.particles.push(new Particle(x, y, color));
        }
    }

    update() {
        this.particles = this.particles.filter(p => p.life > 0);
        this.particles.forEach(p => p.update());
    }

    draw(ctx) {
        this.particles.forEach(p => p.draw(ctx));
    }
}

const particleSystem = new ParticleSystem();
```

```javascript
class Boss extends Entity {
    constructor(x, y, type) {
        super(x, y, 48, 48, '#800080');
        this.type = type;
        this.health = 3;
        this.maxHealth = 3;
        this.phase = 1;
        this.attackTimer = 0;
        this.projectiles = [];
    }

    update() {
        // Boss AI
        this.attackTimer++;
        
        if (this.attackTimer >= 60) {
            this.attack();
            this.attackTimer = 0;
        }
        
        // 更新投射物
        this.projectiles = this.projectiles.filter(p => p.life > 0);
        this.projectiles.forEach(p => {
            p.x += p.speedX;
            p.y += p.speedY;
            p.life--;
        });
    }

    attack() {
        // 发射投射物
        this.projectiles.push({
            x: this.x + 24,
            y: this.y + 24,
            speedX: -3,
            speedY: 0,
            life: 60,
            color: '#ff0000'
        });
    }

    takeDamage() {
        this.health--;
        screenShake.shake(15, 20);
        particleSystem.emit(this.x + 24, this.y + 24, '#800080', 20);
        
        if (this.health <= 0) {
            return true; // Boss被击败
        }
        return false;
    }

    draw(ctx) {
        // 绘制Boss身体
        ctx.fillStyle = this.color;
        ctx.fillRect(this.x, this.y, 48, 48);
        
        // 绘制Boss眼睛
        ctx.fillStyle = '#ff0000';
        ctx.fillRect(this.x + 10, this.y + 10, 8, 8);
        ctx.fillRect(this.x + 30, this.y + 10, 8, 8);
        
        // 绘制血条
        ctx.fillStyle = '#333';
        ctx.fillRect(this.x, this.y - 10, 48, 6);
        ctx.fillStyle = '#ff0000';
        ctx.fillRect(this.x, this.y - 10, 48 * (this.health / this.maxHealth), 6);
        
        // 绘制投射物
        this.projectiles.forEach(p => {
            ctx.fillStyle = p.color;
            ctx.fillRect(p.x, p.y, 8, 8);
        });
    }
}
```
